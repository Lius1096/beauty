  <script>
    // MOBILE MENU TOGGLE
    const mobileMenuBtn = document.getElementById('mobileMenuBtn');
    const mobileMenu = document.getElementById('mobileMenu');
    if (mobileMenuBtn && mobileMenu) {
      mobileMenuBtn.addEventListener('click', () => {
        mobileMenu.classList.toggle('hidden');
      });
    }

    // MODAL OPEN / CLOSE
    const modal = document.getElementById('modal');
    const openModalBtns = [
      document.getElementById('openModalBtn'),
      document.getElementById('openModalBtnHero'),
      document.getElementById('openModalBtnMobile')
    ].filter(Boolean); // filtre les boutons qui existent
    const closeModalBtn = document.getElementById('closeModalBtn');

    openModalBtns.forEach(btn => {
      btn.addEventListener('click', () => {
        modal.classList.remove('hidden');
        document.body.style.overflow = 'hidden'; // bloquer le scroll quand modal ouvert
      });
    });

    function resetDateMin() {
      const today = new Date().toISOString().split('T')[0];
      const dateInput = document.getElementById('date');
      if (dateInput) dateInput.min = today;
    }

    function clearMessage() {
      const messageEl = document.getElementById('message');
      if (messageEl) {
        messageEl.textContent = '';
        messageEl.classList.remove('text-green-600', 'text-red-600');
      }
    }

    if (closeModalBtn && modal) {
      closeModalBtn.addEventListener('click', () => {
        modal.classList.add('hidden');
        document.body.style.overflow = 'auto'; // autoriser le scroll
        clearMessage();
        const form = document.getElementById('appointmentForm');
        if (form) form.reset();
        resetDateMin();
      });
    }

    if (modal) {
      modal.addEventListener('click', (e) => {
        if (e.target === modal) {
          modal.classList.add('hidden');
          document.body.style.overflow = 'auto';
          clearMessage();
          const form = document.getElementById('appointmentForm');
          if (form) form.reset();
          resetDateMin();
        }
      });
    }

    // FORM LOGIC
    const form = document.getElementById('appointmentForm');
    if (form) {
      // Initialiser date min à aujourd'hui
      resetDateMin();

      form.addEventListener('submit', async (e) => {
        e.preventDefault();
        clearMessage();

        const name = form.clientName.value.trim();
        const phone = form.clientPhone.value.trim();
        const technician = form.technician.value;
        const service = form.service.value;
        const date = form.date.value;
        const startTime = form.startTime.value;
        const endTime = form.endTime.value;
        const messageEl = document.getElementById('message');

        // Validation basique
        if (!name || !phone || !technician || !service || !date || !startTime || !endTime) {
          if (messageEl) {
            messageEl.textContent = 'Veuillez remplir tous les champs obligatoires.';
            messageEl.classList.add('text-red-600');
          }
          return;
        }

        if (startTime >= endTime) {
          if (messageEl) {
            messageEl.textContent = "L'heure de fin doit être supérieure à l'heure de début.";
            messageEl.classList.add('text-red-600');
          }
          return;
        }

        // Construire des ISO dates complètes
        const startDateTime = new Date(`${date}T${startTime}:00`);
        const endDateTime = new Date(`${date}T${endTime}:00`);

        // Appel API vers ton backend
        try {
          const response = await fetch('http://localhost:5000/appointments/', {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
            },
            body: JSON.stringify({
              clientName: name,
              clientPhone: phone,
              technician,
              service,
              startTime: startDateTime.toISOString(),
              endTime: endDateTime.toISOString()
            }),
          });

          const data = await response.json();

          if (response.ok && data.success) {
            if (messageEl) {
              messageEl.textContent = 'Réservation effectuée avec succès ! Merci 😊';
              messageEl.classList.add('text-green-600');
            }

            setTimeout(() => {
              form.reset();
              clearMessage();
              modal.classList.add('hidden');
              document.body.style.overflow = 'auto';
              resetDateMin();
            }, 3000);
          } else {
            if (messageEl) {
              messageEl.textContent = data.message || 'Erreur lors de la réservation.';
              messageEl.classList.add('text-red-600');
            }
          }
        } catch (error) {
          if (messageEl) {
            messageEl.textContent = 'Erreur réseau ou serveur. Merci de réessayer plus tard.';
            messageEl.classList.add('text-red-600');
          }
          console.error('Erreur API:', error);
        }
      });
    }
  </script>

      if (data.success && Array.isArray(data.data)) {
        data.data.forEach(image => {
          const img = document.createElement("img");

          img.alt = "Portfolio";
          img.className = "rounded-lg shadow hover:scale-105 transition-transform duration-300 cursor-pointer";
          galleryContainer.appendChild(img);
        });
      }
    })

    });
});
